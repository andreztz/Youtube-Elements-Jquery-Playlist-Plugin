Features & Configration

Setting Up
						<link rel="stylesheet" href="youtubeElements/youtubeElements.css" />
						<script src="https://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min.js"></script>
						<script type="text/javascript" src="youtubeElements/youtubeElements.js"></script>
						<script type="text/javascript">
							$(function(){
								$('#videoPlayer').youtubeElements({
									playlistId: "11D2683FF69C635F",
									defaultPath: "youtubeElements",
									playerSkin:"custom",
									playerWidth: "640",
									playerHeight: "360",
									autoPlay:true
								});
							});
						</script>
